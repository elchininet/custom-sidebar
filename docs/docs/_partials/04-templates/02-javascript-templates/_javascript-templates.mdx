      [[[
        return 'Info (' + state_attr('update.home_assistant_supervisor_update', 'latest_version') + ')';
      ]]]

      [[[
        return 'OS ' + state_attr('update.home_assistant_operating_system_update', 'latest_version');
      ]]]
    href: '/config/info'
    icon: 'mdi:information-outline'

    item: 'restart'

        if (user_is_admin) {
          openRestartDialog();
