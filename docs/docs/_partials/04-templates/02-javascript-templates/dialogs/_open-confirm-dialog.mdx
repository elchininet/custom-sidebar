```typescript
openConfirmDialog({
  // The title of the dialog
  title: string;
  // The text of the dialog
  text: string;
  // The text of the confirm button
  confirmText?: string;
  // The text of the dismiss button
  dismissText?: string;
  // Indicates if the confirm dialog is destructive
  destructive?: boolean;
  // Function that will be executed when the confirm button is clicked
  confirm?: () => void;
  // Function that will be executed when the dismiss button is clicked
  cancel?: () => void;
})
```
